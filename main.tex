\begin{align} \label{eq:change_basis}
	\vec{OP} = x \vec{u}_x + y \vec{u}_y = (x \cos \theta + y \sin \theta) \vec{u}_x^* + (y \cos \theta - x \sin \theta) \vec{u}_y^*.
\end{align}
